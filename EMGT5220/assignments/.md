	[*] --> Install
	state Install {
		install_sv : Install server host machine
        [*] --> install_sv
        state unit_test_sv <<choice>>
        install_sv --> unit_test_sv
        unit_test_sv --> [*]
        replace_sv : Replace failing components
        unit_test_sv --> replace_sv : Unit test failed
        replace_sv --> install_sv : Reinstall
        --
        install_cb : Install cable
        [*] --> install_cb
        state unit_test_cb <<choice>>
        install_cb --> unit_test_cb
        unit_test_cb --> [*]
        replace_cb : Replace failing components
        unit_test_cb --> replace_cb : Unit test failed
        replace_cb --> install_cb : Reinstall
        --
        install_tm : Install terminal devices
        [*] --> install_tm
        state unit_test_tm <<choice>>
        install_tm --> unit_test_tm
        unit_test_tm --> [*]
        replace_tm : Replace failing components
        unit_test_tm --> replace_tm : Unit test failed
        replace_tm --> install_tm : Reinstall
    }
    Install --> Connect
    Connect --> Configure
 	State Configure {
		customize : Customize operating system configuration
		make_image : Make operating system image
		[*] --> customize
		customize --> make_image
		make_image --> [*]
 	}
 	Configure --> Deploy
 	state Deploy {
 		instantiate : Instantiate virtual machines for every employee with system image
 		distribute : assign virtual machines to employees
 		[*] --> instantiate
 		instantiate --> distribute
 		distribute --> [*]
 	}
 	Deploy --> Maintain
 	state Maintain {
 		[*] --> monitor
 		monitor : Monitor virtual machines' and server's statuses
        state has_error <<choice>>
 		monitor --> has_error
 		has_error --> Fix : Error occurs
 		has_error --> monitor : Funtion well
 		Fix --> monitor
 		monitor --> backup
        backup : Periodically backup
 		backup --> monitor
 	}
